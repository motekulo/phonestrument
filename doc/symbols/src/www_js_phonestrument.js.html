 21 /**
 22  * A Phonestrument is the main controller for the application. It draws the
 23  * main (home) screen, and controls how components are connected together. Most
 24  * of the action takes place in the callbacks from the Interface widgets.
 25  *
 26  **/
 27 function Phonestrument(){
 28 
 29     var monosynth = new Basicmonosynth();
 30     var barseq = new BarSequencer();
 31     var part = new Part();
 32     var adaptor = null;
 33     var currentbarnum = 0;
 34     var currentpos = 0;
 35 
 36     /**
 37      * Schedule a regular start of bar notification to a bar sequencer, so that
 38      * it can draw itself
 39      *
 40      **/
 41     Tone.Transport.scheduleRepeat(function(time){
 42             //console.log("Bar: " + currentbarnum);
 43             currentpos = Tone.Transport.position;
 44             console.log("Bar: " + currentpos);
 45             barseq.setCurrentBarNum(currentpos);
 46             currentbarnum++;
 47     }, "1m");
 48 
 49     Tone.Transport.loop = true;
 50     Tone.Transport.loopStart = 0;
 51     Tone.Transport.loopEnd = "4m";
 52 
 53     var mp = new Interface.Panel({ 
 54         container:document.querySelector("#InterfacePanel") 
 55     });
 56 
 57     var home = new Interface.Button({
 58         bounds: [0, 0, .2, .1],
 59         label: "Home",
 60         mode: 'momentary',
 61         onmouseup : function() {
 62             mp.clear();
 63             mp.add(home, ib1, ib2, tb, pb, pbl, dconnect);
 64             barseq.setForeground(false); // stop sequencer from redrawing its view
 65 
 66         }
 67     });
 68 
 69 
 70     var ib1 = new Interface.Button({
 71         bounds: [.25, 0, .2, .1],
 72         label: "Monosynth",
 73         mode: 'momentary',
 74         onmouseup : function() {
 75             mp.remove(ib1);
 76             mp.remove(ib2);
 77             mp.remove(tb);
 78             mp.remove(pb);
 79             mp.remove(pbl);
 80             mp.remove(dconnect);
 81             monosynth.draw(mp);
 82             barseq.setForeground(false); // stop sequencer from redrawing its view
 83         } 
 84 
 85     }
 86     );
 87 
 88     var ib2 = new Interface.Button({
 89         bounds: [.5, 0, .2, .1],
 90         label: "Bar seq",
 91         mode: 'momentary',
 92         onmouseup : function() {
 93             mp.remove(ib1);
 94             mp.remove(ib2);
 95             mp.remove(tb);
 96             mp.remove(pb);
 97             mp.remove(pbl);
 98             mp.remove(dconnect);
 99             barseq.setForeground(true); // stop sequencer from redrawing its view
100             barseq.draw(mp);
101         } 
102 
103     });
104 
105     var tb = new Interface.ButtonV({
106         bounds: [.05, .15, .05, .1],
107         points: [{x:0,y:0},{x:1,y:.5},{x:0,y:1}],
108         mode: 'toggle', 
109         textLocation: {x:.5,y:.5},
110         onvaluechange : function (){
111             if (this.value == 1) {
112                 console.log("Transport start");
113                 Tone.Transport.start();
114             } else {
115                 console.log("Transport stop");
116                 Tone.Transport.stop();
117             }
118         }
119     });
120 
121     var tfl_bar = new Interface.Label({ 
122         bounds:[.25,.15, .1, .1],
123         size: 24,
124         style: 'bold',
125         hAlign:"left",
126         value: "0"
127     });
128 
129     var pb = new Interface.Patchbay({ 
130         bounds:[0, .25, 1, .5],
131         points:[ 
132         {name:'syn-in'},{name:'syn-out'},
133         {name:'seq-in'},{name:'seq-out'},
134         {name:'pt1-in'},{name:'pt1-out'}, 
135         {name:'main', name2:'in' }],
136         onconnection: function( start, end ) {
137             pbl.setValue( start.name + ' connected to ' + end.name );
138             console.log("Connection made " + start.name + 
139                     " " + start.name2 + " to " + end.name + 
140                     " " + end.name2);
141 
142             if ((start.name == "syn-out") && (end.name == "main")) {
143                 monosynth.connectSynth();
144             };
145 
146             if ((start.name == "pt1-out") && (end.name == "syn-in")) {
147                 console.log("going to connect pt1 synth to syn-in");
148                 part.setSynthOut(monosynth.getSynth());
149             };
150 
151             if ((start.name == "seq-out") && (end.name == "pt1-in")) {
152                 console.log("connecting seq out to pt1 in");
153                 // we are connecting an adaptor between the sequence and part
154                 adaptor = new SimpleBarSequencerAdaptor();
155                 adaptor.connectToPart(part);
156                 barseq.setAdaptor(adaptor);
157                 barseq.initExternalPart(part);
158                 barseq.isConnected = true;
159             };
160 
161 
162             if ((start.name == "seq-out") && (end.name == "syn-in")) {
163                 console.log("connecting seq out to syn in");
164                 barseq.setSynthOut(monosynth.getSynth);
165                 barseq.initPart();
166             };
167 
168         },
169         ondisconnection: function( start, end ) {
170             pbl.setValue( start.name + ' disconnected from ' + end.name );
171             console.log("Connection made");
172             if ((start.name == "syn-out") && (end.name == "main")) {
173                 monosynth.disconnectSynth();
174             };
175 
176 
177         }
178     }); 
179 
180     var pbl = new Interface.Label({
181         bounds:[.05, .75, 1, .05],
182         vAlign:'middle',
183         value:'Drag a cable between patch points',
184     })
185 
186     var dconnect = new Interface.Button({
187         bounds: [0, .8, .1, .1],
188         label: "Disconnect",
189         mode: 'momentary',
190         onmouseup : function() {
191             for (i=0; i < pb.connections.length; i++) {
192                 console.log("connection " + pb.connections[i]);
193                 pb.deleteConnection(pb.connections[i]);
194             }
195         }
196     });
197 
198     mp.add(home, ib1, ib2, tb, tfl_bar, pb, pbl, dconnect);
199 
200 }
