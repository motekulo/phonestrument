 21 function SimpleBarSequencerAdaptor() {
 22     var octave = 4; // fudge for now
 23     var pitch = ["C","D","E","F"]; // fudge it for now
 24 
 25     // converts from:
 26     // Array of data in row, column, value form
 27     //
 28     // converts to:
 29     // voice index, time (eg "1 * 4s"), pitch (eg "C4)
 30     //
 31 
 32     this.part = null;
 33 
 34     this.connectToPart = function(part) {
 35         this.part = part;
 36 
 37     }
 38 
 39     this.convertData = function(data) {
 40         /* From a simple bar sequencer we get position (transport), row, column, value data
 41 
 42            A part expects a voice index (row, in this case), time, and pitch
 43 
 44 
 45 
 46 */
 47         if (data.length >= 4) {
 48             var pos = data[0];
 49             var row = data[1];
 50             var col = data[2];
 51             var val = data[3];
 52 
 53             var index = row;
 54             //var time = col + " * 16n";
 55             var time = pos + " + (" + col + " * 16n)";
 56             console.log("Time is " + time);
 57             if (val == 1) {
 58                 note = pitch[row] + octave;
 59             } else {
 60                 note = null;
 61             }
 62             var converteddata = [index, time, note];
 63             return converteddata;
 64         }
 65 
 66     }
 67 
 68     this.sendConvertedDataToPart = function(data) {
 69         var index = data[0]; //FIXME indexed collection would be better
 70         var time = data[1];
 71         var note = data[2];
 72         this.part.setNoteArray(index, time, note);
 73     }
 74 
 75     /** Queries associated part for note data for a particular bar,
 76      * then returns an array for the associated sequencer for
 77      * display purposes (so bararray[1][4]==0 means there is no
 78      * note to display on voice 1, 5th indicator/button on
 79      * sequencer).
 80      *
 81      * @param {int} bar - The bar number to query 
 82      * @param {division} division - The subdivision level to
 83      * query (so 16 is 16th note)
 84      * @returns - a 2 dim array of voices and notes (vaues 1 or 0)  
 85      **/
 86 
 87     this.getBarArray = function(bar, division){
 88         var bararray = [];
 89         var note;
 90         var time;
 91         for (j=0; j < this.part.voices.length; j++){
 92             var voicearray = [];
 93             for (i=0; i < division;i++){
 94                 time = bar + "m" + " + (" + i + " * " + division + "n)";
 95                 note = this.part.voices[j].at(time);
 96                 if (note != null) {
 97                     voicearray[i] = 1;
 98                 } else {
 99                     voicearray[i] = 0;
100                 }
101                 //return time;
102 
103 
104             }
105             bararray[j] = voicearray;
106         }
107         return bararray;
108 
109     }
110 
111 }
